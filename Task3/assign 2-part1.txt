  <script>
    $(document).ready(function(){

      function toggleCardFields() {
        const selected = $('input[name="paymentMethod"]:checked').val();
        if(selected === 'card') {
          $('#cardFields').show().find('input, select').prop('disabled', false);
        } else {
          $('#cardFields').hide().find('input, select').prop('disabled', true);
        }
      }

      $('input[name="paymentMethod"]').change(toggleCardFields);
      toggleCardFields();

      $('#checkout-form').submit(function(e){
        e.preventDefault();
        let valid = true;

        // Full Name
        const fullName = $('#fullName').val().trim();
        if(fullName.length < 3){
          $('#fullName').addClass('is-invalid').removeClass('is-valid');
          valid = false;
        } else { $('#fullName').addClass('is-valid').removeClass('is-invalid'); }

        // Email
        const email = $('#email').val().trim();
        const emailReg = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
        if(!emailReg.test(email)){
          $('#email').addClass('is-invalid').removeClass('is-valid');
          valid = false;
        } else { $('#email').addClass('is-valid').removeClass('is-invalid'); }

        // Phone
        const phone = $('#phone').val().trim();
        const phoneReg = /^\d{10,}$/;
        if(!phoneReg.test(phone)){
          $('#phone').addClass('is-invalid').removeClass('is-valid');
          valid = false;
        } else { $('#phone').addClass('is-valid').removeClass('is-invalid'); }

        // Address
        const address = $('#address').val().trim();
        if(address===''){ $('#address').addClass('is-invalid').removeClass('is-valid'); valid=false;}
        else {$('#address').addClass('is-valid').removeClass('is-invalid');}

        // City
        const city = $('#city').val().trim();
        if(city===''){ $('#city').addClass('is-invalid').removeClass('is-valid'); valid=false;}
        else {$('#city').addClass('is-valid').removeClass('is-invalid');}

        // Postal
        const postal = $('#postal').val().trim();
        const postalReg = /^\d{4,6}$/;
        if(!postalReg.test(postal)){ $('#postal').addClass('is-invalid').removeClass('is-valid'); valid=false;}
        else {$('#postal').addClass('is-valid').removeClass('is-invalid');}

        // Country
        if($('#country').val()===''){ $('#country').addClass('is-invalid').removeClass('is-valid'); valid=false;}
        else {$('#country').addClass('is-valid').removeClass('is-invalid');}

        // Payment method
        if($('input[name="paymentMethod"]:checked').length===0){ valid=false; alert("Select a payment method"); }

        // Card Fields if Card selected
        if($('input[name="paymentMethod"]:checked').val() === 'card'){
          const cardName = $('#cardName').val().trim();
          if(cardName===''){ $('#cardName').addClass('is-invalid').removeClass('is-valid'); valid=false;} 
          else {$('#cardName').addClass('is-valid').removeClass('is-invalid');}

          const cardNumber = $('#cardNumber').val().trim();
          const cardReg = /^\d{13,19}$/;
          if(!cardReg.test(cardNumber)){ $('#cardNumber').addClass('is-invalid').removeClass('is-valid'); valid=false;} 
          else {$('#cardNumber').addClass('is-valid').removeClass('is-invalid');}

          const expiry = $('#expiry').val().trim();
          const expiryReg = /^(0[1-9]|1[0-2])\/\d{2}$/;
          if(!expiryReg.test(expiry)){ $('#expiry').addClass('is-invalid').removeClass('is-valid'); valid=false;} 
          else {$('#expiry').addClass('is-valid').removeClass('is-invalid');}

          const cvv = $('#cvv').val().trim();
          const cvvReg = /^\d{3,4}$/;
          if(!cvvReg.test(cvv)){ $('#cvv').addClass('is-invalid').removeClass('is-valid'); valid=false;} 
          else {$('#cvv').addClass('is-valid').removeClass('is-invalid');}
        }

        // Terms checkbox
        if(!$('#terms').is(':checked')){ $('#terms').addClass('is-invalid'); valid=false;}
        else { $('#terms').removeClass('is-invalid'); }

        // Scroll to first invalid
        if(!valid){
          $('html, body').animate({
            scrollTop: $('.is-invalid').first().offset().top - 80
          }, 500);
          return false;
        }

        alert("Order placed successfully (demo)!");
      });
    });
  </script>
