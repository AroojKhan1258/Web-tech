<script>
$(document).ready(function(){

  // Sample Cart Items
  let cart = [
    { id:1, name:"Skydiving Experience", price:12000, quantity:1 },
    { id:2, name:"Mountain Climb", price:12000, quantity:2 }
  ];
  const shippingCost = 500;
  let taxRate = 0.1;

  function renderCart(){
    let subtotal = 0;
    $("#cartItems").empty();
    cart.forEach(item=>{
      let totalPrice = item.price * item.quantity;
      subtotal += totalPrice;
      $("#cartItems").append(`
        <li class="list-group-item">
          <div>
            <strong>${item.name}</strong><br>
            <span class="muted">₨ ${item.price} x </span>
            <input type="number" class="quantity-input" min="1" data-id="${item.id}" value="${item.quantity}">
          </div>
          <div>
            <span>₨ ${totalPrice}</span>
            <button class="btn btn-sm btn-danger btn-remove" data-id="${item.id}">Remove</button>
          </div>
        </li>
      `);
    });
    let tax = Math.round(subtotal * taxRate);
    let grandTotal = subtotal + shippingCost + tax;
    $("#subtotal").text("₨ " + subtotal);
    $("#tax").text("₨ " + tax);
    $("#grandTotal").text("₨ " + grandTotal);
  }
  renderCart();

  // Quantity change
  $(document).on("input", ".quantity-input", function(){
    let id = $(this).data("id");
    let value = parseInt($(this).val());
    if(value<1) value=1;
    cart = cart.map(item=> item.id===id ? {...item, quantity:value} : item);
    renderCart();
  });

  // Remove item
  $(document).on("click", ".btn-remove", function(){
    let id = $(this).data("id");
    cart = cart.filter(item=>item.id!==id);
    renderCart();
  });

  // Coupon
  $("#applyCoupon").click(function(){
    let code = $("#couponCode").val().trim();
    if(code.toLowerCase()==="discount10"){
      alert("Coupon applied! 10% off subtotal");
      cart = cart.map(item=>({...item, price:Math.round(item.price*0.9)}));
      renderCart();
    } else { alert("Invalid coupon"); }
  });

  // Step navigation
  $(".step").click(function(){
    let step = $(this).data("step");
    $(".step").removeClass("active");
    $(this).addClass("active");
    $(".checkout-section").hide();
    if(step==="cart") $("#cartSection").show();
    else if(step==="customer") $("#customerSection").show();
    else if(step==="payment") $("#paymentSection").show();
    else if(step==="review") $("#reviewSection").show();
  });

  // Payment fields toggle
  function toggleCardFields(){
    let selected = $('input[name="paymentMethod"]:checked').val();
    if(selected==='card') $("#cardFields").show().find('input,select').prop('disabled',false);
    else $("#cardFields").hide().find('input,select').prop('disabled',true);
  }
  $('input[name="paymentMethod"]').change(toggleCardFields);
  toggleCardFields();

  // Form validation
  function validateForm(){
    let valid = true;
    if($("#fullName").val().trim().length<3){ $("#fullName").addClass("is-invalid"); valid=false; } else { $("#fullName").removeClass("is-invalid"); }
    const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if(!emailPattern.test($("#email").val().trim())){ $("#email").addClass("is-invalid"); valid=false; } else { $("#email").removeClass("is-invalid"); }
    if(!/^\d{10,}$/.test($("#phone").val().trim())){ $("#phone").addClass("is-invalid"); valid=false; } else { $("#phone").removeClass("is-invalid"); }
    if($("#address").val().trim()===""){ $("#address").addClass("is-invalid"); valid=false; } else { $("#address").removeClass("is-invalid"); }
    if($("#city").val().trim()===""){ $("#city").addClass("is-invalid"); valid=false; } else { $("#city").removeClass("is-invalid"); }
    if(!/^\d{4,6}$/.test($("#postal").val().trim())){ $("#postal").addClass("is-invalid"); valid=false; } else { $("#postal").removeClass("is-invalid"); }
    if($("#country").val()===""){ $("#country").addClass("is-invalid"); valid=false; } else { $("#country").removeClass("is-invalid"); }
    if(!$('input[name="paymentMethod"]:checked').val()){ $("#paymentError").show(); valid=false; } else { $("#paymentError").hide(); }
    if($('input[name="paymentMethod"]:checked').val()==="card"){
      ["#cardName","#cardNumber","#expiry","#cvv"].forEach(id=>{
        if($(id).val().trim()===""){ $(id).addClass("is-invalid"); valid=false; } else { $(id).removeClass("is-invalid"); }
      });
    }
    if(!$("#terms").prop("checked")){ $("#termsError").show(); valid=false; } else { $("#termsError").hide(); }
    return valid;
  }

  // Place Order
  $("#placeOrder").click(function(e){
    e.preventDefault();
    if(validateForm()){
      alert("Order placed successfully!");
      $(".checkout-section").hide();
      $("#reviewSection").show();
      $(".step").removeClass("active");
      $(".step[data-step='review']").addClass("active");
    }
  });

});

<script>
$(document).ready(function(){

  const apiURL = "https://fakestoreapi.com/products";
  let items = [];
  let editingId = null;

  function renderList(){
    $("#itemList").empty();
    if(items.length===0){ $("#itemList").append("<li>No items available.</li>"); return; }
    items.forEach(item=>{
      $("#itemList").append(`
        <li class="list-group-item d-flex justify-content-between align-items-center">
          <div>
            <strong>${item.title}</strong> - ₨ ${item.price}
          </div>
          <div>
            <button class="btn btn-sm btn-warning edit" data-id="${item.id}">Edit</button>
            <button class="btn btn-sm btn-danger remove" data-id="${item.id}">Delete</button>
          </div>
        </li>
      `);
    });
  }

  // Load items from API
  function loadItems(){
    $("#loading").show();
    $("#listError").text("");
    $.ajax({
      url: apiURL,
      method: "GET",
      success: function(data){
        items = data.map(i=>({id:i.id, title:i.title, price:i.price}));
        $("#loading").hide();
        renderList();
      },
      error: function(){
        $("#loading").hide();
        $("#listError").text("Failed to load items. Try again.");
      }
    });
  }

  loadItems();

  // Add/Update item
  $("#addItem").click(function(e){
    e.preventDefault();
    const title = $("#itemName").val().trim();
    const price = parseFloat($("#itemPrice").val());
    if(!title || price<=0){ $("#formError").text("Enter valid name and price."); return; }
    $("#formError").text("");

    const payload = { title, price, description:"Demo item", image:"https://i.pravatar.cc", category:"general" };

    if(editingId){
      // Update item
      $.ajax({
        url: `${apiURL}/${editingId}`,
        method:"PUT",
        data: JSON.stringify(payload),
        contentType:"application/json",
        success: function(data){
          loadItems();
          editingId = null;
          $("#crudForm")[0].reset();
        },
        error: function(){ $("#formError").text("Update failed."); }
      });
    } else {
      // Create new item
      $.ajax({
        url: apiURL,
        method:"POST",
        data: JSON.stringify(payload),
        contentType:"application/json",
        success: function(data){
          loadItems();
          $("#crudForm")[0].reset();
        },
        error: function(){ $("#formError").text("Add failed."); }
      });
    }
  });

  // Delete item
  $(document).on("click",".remove", function(){
    const id = $(this).data("id");
    $.ajax({
      url: `${apiURL}/${id}`,
      method:"DELETE",
      success: function(){ loadItems(); },
      error: function(){ $("#listError").text("Delete failed."); }
    });
  });

  // Edit item
  $(document).on("click",".edit", function(){
    const id = $(this).data("id");
    const item = items.find(i=>i.id===id);
    if(item){
      $("#itemName").val(item.title);
      $("#itemPrice").val(item.price);
      editingId = id;
    }
  });

});
</script>
